definitionsDecoder : Decoder Schema.Definitions
definitionsDecoder =

decoderTests : Test
decoderTests =
    describe "Decoder"
        [ describe "$defs support (DEC-01)"
            [ test "$defs keys are normalized to #/definitions/ prefix" <|
                \_ ->
                    let
                        json = """{"$defs":{"Address":{"type":"object","properties":{"street":{"type":"string"}}}},"type":"object","properties":{"addr":{"$ref":"#/$defs/Address"}}}"""
                        result = decodeString Json.Schema.Decode.decoder json
                    in
                    case result of
                        Ok model ->
                            Dict.member "#/definitions/Address" model.definitions
                                |> Expect.equal True
                        Err e ->
                            Expect.fail (Json.Decode.errorToString e)
            , test "$ref values pointing at $defs are normalized" <|
                \_ ->
                    let
                        json = """{"$defs":{"Foo":{"type":"string"}},"type":"object","properties":{"bar":{"$ref":"#/$defs/Foo"}}}"""
                        result = decodeString Json.Schema.Decode.decoder json
                    in
                    case result of
                        Ok model ->
                            case model.schema of
                                Schema.Object { properties } ->
                                    case properties of
                                        [ Schema.Optional "bar" (Schema.Ref { ref }) ] ->
                                            Expect.equal "#/definitions/Foo" ref
                                        _ ->
                                            Expect.fail "Expected single optional property 'bar' with Ref"
                                _ ->
                                    Expect.fail "Expected Object schema"
                        Err e ->
                            Expect.fail (Json.Decode.errorToString e)
            , test "both definitions and $defs are merged" <|
                \_ ->
                    let
                        json = """{"definitions":{"A":{"type":"string"}},"$defs":{"B":{"type":"integer"}},"type":"object","properties":{}}"""
                        result = decodeString Json.Schema.Decode.decoder json
                    in
                    case result of
                        Ok model ->
                            Expect.all
                                [ \m -> Dict.member "#/definitions/A" m.definitions |> Expect.equal True
                                , \m -> Dict.member "#/definitions/B" m.definitions |> Expect.equal True
                                ]
                                model
                        Err e ->
                            Expect.fail (Json.Decode.errorToString e)
            ]
        , describe "Combined type+combinator (DEC-02)"
            [ test "object with oneOf decodes as Object with combinator" <|
                \_ ->
                    let
                        json = """{"type":"object","properties":{"name":{"type":"string"}},"oneOf":[{"required":["name"]},{"required":[]}]}"""
                        result = decodeString Json.Schema.Decode.decoder json
                    in
                    case result of
                        Ok model ->

                                            Expect.pass
                                        _ ->

                                    Expect.fail "Expected Object schema"

                            Expect.fail (Json.Decode.errorToString e)
