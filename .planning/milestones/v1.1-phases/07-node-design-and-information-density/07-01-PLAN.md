**Step 4: Create `iconForSchema` function in `src/Render/Svg.elm`:**

This is a pure helper that determines the correct Icon from a Schema. Add it after the `Icon` type definition. Also add it to the module `exposing` list.

```elm
iconForSchema : Schema -> Icon
iconForSchema schema =
    case schema of
        Schema.String { format, enum } ->
            case enum of
                Just _ ->
                    IEnum

                Nothing ->
                    case format of
                        Just Schema.Email ->
                            IEmail

                        Just Schema.DateTime ->
                            IDateTime

                        Just Schema.Hostname ->
                            IHostname

                        Just Schema.Ipv4 ->
                            IIpv4

                        Just Schema.Ipv6 ->
                            IIpv6

                        Just Schema.Uri ->
                            IUri

                        Just (Schema.Custom s) ->
                            ICustom s

                        Nothing ->
                            IStr

        Schema.Integer { enum } ->
            case enum of
                Just _ ->
                    IEnum

                Nothing ->
                    IInt

        Schema.Number { enum } ->
            case enum of
                Just _ ->
                    IEnum

                Nothing ->
                    IFloat

        Schema.Boolean { enum } ->
            case enum of
                Just _ ->
                    IEnum

                Nothing ->
                    IBool

        Schema.Object _ ->
            IObject

        Schema.Array _ ->
            IList

        Schema.Null _ ->
            INull

        Schema.Ref { ref } ->
            IRef "*"

        Schema.OneOf _ ->
            IObject

        Schema.AnyOf _ ->
            IObject

        Schema.AllOf _ ->
            IObject

        Schema.Fallback _ ->
            INull
```
